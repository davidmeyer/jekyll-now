


\text{Yield}_{\text{\tiny{(zc,n)}}} 
=\Bigg (\bigg (
\frac{\text{Par Value}}{\text{Current Price}} 
\bigg )^{\frac{1}{n}}
 - 1  \Bigg ) * 100




\text{Current Price}_{\text{\tiny{(zc,n)}}}
= \frac{\text{Par Value}} {\bigg ( 1 + 
\Big (\tiny{\frac{\text{Yield}}{100} \Big )} \bigg  )^n}




\text{Current Price}_{\text{\tiny{(zc,1)}}} 
= \frac{\text{Par Value}} { 1 + \tiny{\frac{\text{Yield}}{100}}}




\text{Yield}_{\text{\tiny{(zc,1)}}} 
= \Bigg ( \frac{\text{Par Value} - \text{Current Price}}{\text{Current Price}} \Bigg) * 100



\text{Yield}_{\text{\tiny{zero coupon}}} 
= \Bigg (\frac{\text{Par Value}}{\text{Current Price}} \Bigg)^{\frac{1}{n}} - 1
